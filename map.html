Where in the World?

const myKey = 'YOUR_API_KEY_HERE'; // <--- Put your key here! const map = L.map('map').setView([20, 0], 2); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); async function findMe() { const status = document.getElementById('status'); status.textContent = "Locating..."; navigator.geolocation.getCurrentPosition(async (position) => { const lat = position.coords.latitude; const lon = position.coords.longitude; // 1. Move the map map.setView([lat, lon], 13); // 2. Fetch the weather try { const response = await fetch(`https://api.openweathermap.org/data/2.5/weather?lat=${lat}&lon=${lon}&appid=${myKey}&units=metric`); const data = await response.json(); const temp = data.main.temp; const desc = data.weather[0].description; const icon = data.weather[0].icon; // 3. Drop a marker with the weather info const weatherHTML = `

${temp}°C
${desc}
`; L.marker([lat, lon]).addTo(map) .bindPopup(weatherHTML) .openPopup(); status.textContent = `It's ${temp}°C in your area!`; } catch (error) { status.textContent = "Weather data failed to load."; console.error(error); } }); }
